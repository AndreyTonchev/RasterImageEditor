162 std::vector<std::vector<PixelType>> newPixels(newHeight, std::vector<PixelType>(newWidth));
163
164 std::size_t minWidth = (width > newWidth) ?
165 newWidth : width;
166
167 std::size_t minHeight = (height > newHeight) ?
168 newHeight : height;
169
170
171 for (std::size_t y = 0; y < minHeight; ++y) {
172 for (std::size_t x = 0; x < minWidth; ++x) {
173 newPixels[y][x] = pixels[y][x];
174 }
175 }
176
177 PixelType blank;
178
179 for (std::size_t y = 0; y < newHeight; ++y) {
180 for (std::size_t x = 0; x < newWidth; ++x) {
181 if (y >= height || x >= width) {
182 newPixels[y][x] = blank;
183 }
184 }
185 }
186
187 pixels = std::move(newPixels);
188 width = newWidth;
189 height = newHeight;
